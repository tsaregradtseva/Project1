#include <iostream>
using namespace ::std;

int Summa(int j, int k)
{
	int S = 0;
	int i = j;
	while (i <= k)
	{
		S += 2 * i;
		i++;
	}
	return S;
}

void main()
{
	system("chcp 1251");
	cout << "Введите верхний предел произведения ";
	int k, S, i;
	cin >> k;
	double F = 1;
	//внешний цикл для вычисления произведения
	for (int j = 1; j <= k; j++)
	{
		F *= (double)j / (j + 2) * Summa(j, k);
	}
	cout << "F = " << F << endl;
	system("pause");
}
