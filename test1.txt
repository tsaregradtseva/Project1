#include <iostream>
using namespace ::std;

int Summa(int j, int k)
{
	//цикл для вычисления внутренней суммы
	int S = 0;
	int i = j;
	while (i <= k)
	{
		S += 2 * i;
		i++;
	}
	return S;
}
